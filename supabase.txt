-- =============================================
-- MULTI-LAYER SECURITY CONFIGURATION
-- LAYER 1: Database Constraints
-- LAYER 2: Row Level Security (RLS)
-- LAYER 3: Auth Schema Integration
-- LAYER 4: Admin Role Boundaries
-- =============================================

create extension if not exists pgcrypto;

-- =============================================
-- ADMIN PASSWORD RESET RPC
-- =============================================
create or replace function public.admin_reset_password(user_id uuid, new_password text)
returns void as $$
begin
  -- Check if caller is admin
  if not exists (select 1 from public.users where id = auth.uid() and is_admin = true) then
    raise exception 'Unauthorized';
  end if;
  
  -- Update the password securely
  update auth.users
  set encrypted_password = extensions.crypt(new_password, extensions.gen_salt('bf'))
  where id = user_id;
end;
$$ language plpgsql security definer set search_path = public;

create or replace function public.admin_delete_user(target_user_id uuid)
returns void as $$
begin
  if not exists (select 1 from public.users where id = auth.uid() and is_admin = true) then
    raise exception 'Unauthorized';
  end if;

  if target_user_id = auth.uid() then
    raise exception 'You cannot delete your own active account.';
  end if;

  delete from public.dragon_accounts where user_id = target_user_id;

  if exists (
    select 1
    from information_schema.tables
    where table_schema = 'public'
      and table_name = 'tickets'
  ) then
    execute 'delete from public.tickets where user_id = $1' using target_user_id;
  end if;

  if exists (
    select 1
    from information_schema.tables
    where table_schema = 'public'
      and table_name = 'pets'
  ) then
    execute 'delete from public.pets where user_id = $1' using target_user_id;
  end if;

  delete from public.users where id = target_user_id;
  delete from auth.users where id = target_user_id;
end;
$$ language plpgsql security definer set search_path = public, auth;

-- =============================================
-- USERS TABLE (Linked to Auth.Users)
-- =============================================
-- This table stores profile data for every registered account.
-- Link: public.users.id -> auth.users.id
create table if not exists public.users (
  id uuid primary key references auth.users(id) on delete cascade,
  username text,
  is_admin boolean not null default false,
  created_at timestamptz not null default now()
);

alter table public.users enable row level security;

-- SECURITY: Users can only read their own profile.
drop policy if exists "Users can view their own profile" on public.users;
create policy "Users can view their own profile"
  on public.users for select
  using ( auth.uid() = id );

-- SECURITY: Admins can view all user profiles for management.
-- We use a security definer function to avoid infinite recursion.
-- To be safe from old profiles table:
drop table if exists public.profiles cascade;

create or replace function public.is_admin()
returns boolean as $$
begin
  return exists (
    select 1 from public.users
    where id = auth.uid() and is_admin = true
  );
end;
$$ language plpgsql security definer set search_path = public;

drop policy if exists "Admins can view all users" on public.users;
create policy "Admins can view all users"
  on public.users for select
  using ( public.is_admin() );

-- SECURITY: Users can only update their own profile.
drop policy if exists "Users can update their own profile" on public.users;
create policy "Users can update their own profile"
  on public.users for update
  using ( auth.uid() = id );

-- SECURITY: Auto-profile creation via trigger.
drop policy if exists "Users can insert their own profile" on public.users;
create policy "Users can insert their own profile"
  on public.users for insert
  with check ( auth.uid() = id );

-- Auto-create profile on signup
create or replace function public.handle_new_user()
returns trigger language plpgsql security definer as $$
begin
  insert into public.users (id, username)
  values (new.id, split_part(new.email, '@', 1))
  on conflict (id) do nothing;
  return new;
end;
$$;

drop trigger if exists on_auth_user_created on auth.users;
create trigger on_auth_user_created
  after insert on auth.users
  for each row execute function public.handle_new_user();

-- =============================================
-- CLASS TREE TABLE (Admin managed)
-- =============================================
create table if not exists public.class_tree (
  id uuid primary key default gen_random_uuid(),
  label text not null,
  parent_id uuid references public.class_tree(id) on delete cascade,
  sort_order integer not null default 0,
  created_at timestamptz not null default now()
);

alter table public.class_tree enable row level security;

-- Anyone can read (for UI), but only admins can write.
drop policy if exists "Anyone can view class tree" on public.class_tree;
create policy "Anyone can view class tree"
  on public.class_tree for select
  using ( true );

drop policy if exists "Admins can manage class tree" on public.class_tree;
create policy "Admins can manage class tree"
  on public.class_tree for all
  using (
    exists (select 1 from public.users where id = auth.uid() and is_admin = true)
  );

-- =============================================
-- PATCHNOTES TABLE (Admin managed)
-- =============================================
create table if not exists public.patchnotes (
  id uuid primary key default gen_random_uuid(),
  title text not null,
  content text not null,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

alter table public.patchnotes enable row level security;

-- Anyone can read, only admins can write.
drop policy if exists "Anyone can view patchnotes" on public.patchnotes;
create policy "Anyone can view patchnotes"
  on public.patchnotes for select
  using ( true );

drop policy if exists "Admins can manage patchnotes" on public.patchnotes;
create policy "Admins can manage patchnotes"
  on public.patchnotes for all
  using (
    exists (select 1 from public.users where id = auth.uid() and is_admin = true)
  );

-- =============================================
-- DRAGON ACCOUNTS TABLE (The Sensitive Data)
-- =============================================
create table if not exists public.dragon_accounts (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users(id) default auth.uid(),
  account text not null,
  class_base text not null,
  class_path text[] not null,
  class_label text not null,
  level integer not null default 1,
  spam integer not null default 0,
  ticket integer not null default 0,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  constraint dragon_accounts_level_check check (level >= 1),
  constraint dragon_accounts_spam_check check (spam >= 0),
  constraint dragon_accounts_ticket_check check (ticket >= 0)
);

alter table public.dragon_accounts enable row level security;

-- SECURITY LAYER: STRICT OWNERSHIP
-- A user can only see, insert, update, or delete their OWN records.
drop policy if exists "Users can manage their own dragon accounts" on public.dragon_accounts;
create policy "Users can manage their own dragon accounts"
  on public.dragon_accounts for all
  using ( auth.uid() = user_id )
  with check ( auth.uid() = user_id );

-- SECURITY LAYER: ADMIN OVERRIDE (READ ONLY)
-- Admins can view everything for auditing but cannot modify user accounts.
drop policy if exists "Admins can view all dragon accounts" on public.dragon_accounts;
create policy "Admins can view all dragon accounts"
  on public.dragon_accounts for select
  using ( public.is_admin() );

-- Indices for performance and security lookups
create index if not exists dragon_accounts_user_id_idx on public.dragon_accounts (user_id);
